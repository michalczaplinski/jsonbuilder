## key:opts
1.
entry: name:string
ouput:
{
    key: "name",
    opts: {
        type: "string",
    }
}

2.
entry: name:string|min=10|max=20
ouput:
{
    key: "name",
    opts: {
        type: "string",
        min:10,
        max:20,
    }
}

3.
entry: name:default=rolly
ouput:
{
    key: "name",
    opts: {
        default: ["rolly"]
    }
}
4.
entry: name:default=rolly,erick
ouput:
{
    key: "name",
    opts: {
        default: ["rolly","erick"]
    }
}
